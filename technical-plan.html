Why Azure Container Apps (not AKS)

Auto-scales in 60 seconds, pay-per-use, no Kubernetes expertise needed. Scales to zero when idle. Built-in HTTPS, Dapr sidecar optional. Cost: ~60% cheaper than AKS for this workload.

Database — Export from Live Production

Export SCHEMA + SEED DATA from Live MySQL (192.168.0.149)

We need the live database schema (all tables, indexes, stored procedures) plus reference data (categories, settings, payment configs). NOT user data or bids — those are production-sensitive.

# Step 1: Export schema-only from live production DB mysqldump -h 192.168.0.149 -u $DB_USER -p --no-data \ --routines --triggers --events \ mzadqatar > schema_live.sql # Step 2: Export reference/config data (small tables only) mysqldump -h 192.168.0.149 -u $DB_USER -p \ --no-create-info \ mzadqatar categories types sub_types settings \ payment_methods countries cities \ bid_categories bid_types \ > seed_data.sql # Step 3: Import into local Docker MySQL mysql -h 127.0.0.1 -u root -p gov_auction < schema_live.sql mysql -h 127.0.0.1 -u root -p gov_auction < seed_data.sql # Step 4: Run new government migration on top php artisan migrate # Creates 8 new gov tables # Key tables we need from live: # Schema: users1, bid_products, bids, bid_info, auto_bids, # bid_watch_lists, hold_amounts, request_refunds, # lot_questions, categories, types, sub_types, # payment_history, notifications, settings # Seed: categories, types, sub_types, countries, cities, # payment_methods, bid_categories, settings

New Database Tables

-- 1. Government Entities (ministries, departments) CREATE TABLE government_entities ( id INT AUTO_INCREMENT PRIMARY KEY, name_ar VARCHAR(255) NOT NULL, name_en VARCHAR(255) NOT NULL, logo VARCHAR(500), contact_person VARCHAR(255), contact_email VARCHAR(255), contact_phone VARCHAR(20), is_active TINYINT(1) DEFAULT 1, created_at TIMESTAMP, updated_at TIMESTAMP ); -- 2. Entity Users (roles within entities) CREATE TABLE entity_users ( id INT AUTO_INCREMENT PRIMARY KEY, entity_id INT NOT NULL REFERENCES government_entities(id), user_id INT NOT NULL REFERENCES users1(id), role ENUM('committee','entity_admin','entity_operator','viewer'), created_at TIMESTAMP ); -- 3. Auction Pipeline (workflow tracking) CREATE TABLE auction_pipeline ( id INT AUTO_INCREMENT PRIMARY KEY, bid_product_id INT NOT NULL REFERENCES bid_products(id), entity_id INT NOT NULL REFERENCES government_entities(id), stage ENUM('draft','inspection','photography','valuation', 'approval','published','live','completed','cancelled'), assigned_to INT REFERENCES users1(id), stage_entered_at TIMESTAMP, stage_deadline TIMESTAMP, notes TEXT, created_at TIMESTAMP, updated_at TIMESTAMP ); -- 4. Asset Valuations CREATE TABLE asset_valuations ( id INT AUTO_INCREMENT PRIMARY KEY, bid_product_id INT NOT NULL REFERENCES bid_products(id), entity_id INT NOT NULL REFERENCES government_entities(id), estimated_value DECIMAL(12,2), committee_approved_value DECIMAL(12,2), approved_by INT REFERENCES users1(id), approved_at TIMESTAMP, notes TEXT, created_at TIMESTAMP ); -- 5. Disputes CREATE TABLE disputes ( id INT AUTO_INCREMENT PRIMARY KEY, bid_product_id INT NOT NULL, user_id INT NOT NULL, type ENUM('condition_mismatch','delivery_delay','extension_failed','other'), description TEXT, evidence_files JSON, status ENUM('open','investigating','committee_review','resolved','rejected'), resolution_notes TEXT, resolved_by INT, created_at TIMESTAMP, updated_at TIMESTAMP ); -- 6. KYC (post-win identity verification) CREATE TABLE winner_kyc ( id INT AUTO_INCREMENT PRIMARY KEY, user_id INT NOT NULL, bid_product_id INT NOT NULL, full_name VARCHAR(255), id_type ENUM('national_id','passport','company_cr'), id_number VARCHAR(100), id_front_image VARCHAR(500), id_back_image VARCHAR(500), status ENUM('pending','approved','rejected'), reviewed_by INT, created_at TIMESTAMP, updated_at TIMESTAMP ); -- 7. Invoices / Receipts CREATE TABLE invoices ( id INT AUTO_INCREMENT PRIMARY KEY, invoice_number VARCHAR(50) UNIQUE, user_id INT NOT NULL, bid_product_id INT NOT NULL, amount DECIMAL(12,2), deposit_amount DECIMAL(12,2), commission_amount DECIMAL(12,2), status ENUM('draft','issued','paid','overdue','cancelled'), paid_at TIMESTAMP, pdf_url VARCHAR(500), created_at TIMESTAMP, updated_at TIMESTAMP ); -- 8. Audit Log (government compliance) CREATE TABLE audit_log ( id BIGINT AUTO_INCREMENT PRIMARY KEY, user_id INT, action VARCHAR(100), model_type VARCHAR(100), model_id INT, old_values JSON, new_values JSON, ip_address VARCHAR(45), created_at TIMESTAMP, INDEX(action), INDEX(model_type, model_id), INDEX(created_at) );
